#include <string>
#include <vector>
#include <cmath>
#include <iostream>

using namespace std;

// 두 점 사이 거리
int length(vector<int> d1, vector<int> d2);
// 두 점이 바로 옆으로 연결되어있는지 확인
bool isNext(vector<int> d1, vector<int> d2);

int solution(vector<vector<int>> dots) {
    vector<int> wh;
    for(int di = 1; di < dots.size(); di++) {  
        if (isNext(dots[0], dots[di])) {
            wh.push_back(length(dots[0], dots[di]));
        }
    }
    //      width * height
    return wh[0] * wh[1];
}

int length(vector<int> d1, vector<int> d2) {
    int dx = d1[0] - d2[0];
    int dy = d1[1] - d2[1];
    return static_cast<int>(sqrt(dx * dx + dy * dy));
}

bool isNext(vector<int> d1, vector<int> d2) {
    return (d1[0] == d2[0]) || (d1[1] == d2[1]);
}
